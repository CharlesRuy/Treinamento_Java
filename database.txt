CREATE TABLE tb_categoria_curso
(
	ID BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	DS_CATEGORIA VARCHAR(255) NOT NULL
);

CREATE TABLE tb_curso
(
	ID BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	DS_CURSO VARCHAR(255) NOT NULL,
	NR_DURACAO INT NOT NULL,
	IN_PCD BOOL NOT NULL,
	IN_EXCLUIDO BOOL NOT NULL,
	ID_CATEGORIA_CURSO BIGINT NOT NULL,
	CONSTRAINT FK_ID_CATEGORIA_CURSO FOREIGN KEY (ID_CATEGORIA_CURSO) REFERENCES tb_categoria_curso (ID)
);

CREATE TABLE tb_curso_log
(
	ID BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	DS_NOME VARCHAR(255) NOT NULL,
	DT_LOG DATETIME NOT NULL,
	DS_ACAO VARCHAR(50) NOT NULL,
	ID_CURSO BIGINT NOT NULL,
	CONSTRAINT FK_ID_CURSO FOREIGN KEY (ID_CURSO) REFERENCES tb_curso (ID)
);

CREATE TABLE tb_usuario
(
	ID BIGINT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	EMAIL VARCHAR(200) NOT NULL,
	NOME VARCHAR(200) NOT NULL,
	SENHA VARCHAR(200) NOT NULL
);
